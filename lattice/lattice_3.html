def hadamard_ratio(V):
    prod_norm = 1
    n = V.nrows()
    for i in range(n):
        row = V.row(i)
        row_norm = row.norm()  # è®¡ç®—æ¬§å‡ é‡Œå¾—æ°èŒƒæ•°
        prod_norm *= row_norm
    # è®¡ç®— Hadamard æ¯”ç‡
    ratio = abs(V.determinant()) / prod_norm if prod_norm != 0 else 0
    return (ratio)**(1/n).n()
